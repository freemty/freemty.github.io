 4def _discount_and_norm_reward(self): # reward decay and normalize
 5    discount_ep_rs = np.zeros_like(self.ep_rs)
 6    running_add = 0
 7    for t in reversed(range(0, len(self.ep_rs))):
 8        running_add = running_add * self.gamma + self.ep_rs[t]
 9        #蒙特卡洛方法
10        discount_ep_rs[t] = running_add
11
12    # normalize episode rewards
13    discount_ep_rs -= np.mean(discount_ep_rs)
14    discount_ep_rs /= np.std(discount_ep_rs)
15    return discount_ep_rs
